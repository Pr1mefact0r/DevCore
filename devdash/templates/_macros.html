{# ============================================================ Shared Jinja macros — filter bars + small UI fragments. Keep all UX wiring here so every page is consistent. Ported from an upstream dev-page idiom onto DevDash tokens. ============================================================ #} {# Render one filter group. Options is a list of (value, label, count_or_none). `current` is the currently-selected value. `param` is the query-string key to set. `base_qs` is the rest of the query string to preserve (string, joined). #} {% macro filter_group(label, param, current, options, base_qs='') %}
{{ label }} {% for opt in options %} {% set value = opt[0] %} {% set olabel = opt[1] %} {% set count = opt[2] if opt|length > 2 else None %} {{ olabel }}{% if count is not none %} {{ count }}{% endif %} {% endfor %}
{% endmacro %} {# Limit picker — fixed sizes 50/100/200/500. Preserves the rest of the QS. #} {% macro limit_picker(current, base_qs='', sizes=[50,100,200,500]) %}
Limit {% for n in sizes %} {{ n }} {% endfor %}
{% endmacro %} {# Inline count chip — small grouped data badge. #} {% macro count_chip(label, value, tone='') %} {{ label }} {{ value }} {% endmacro %} {# Sub-nav row — sibling-page navigation. `items` is a list of (path, label, count_or_none). `domain` is 'dev' (controls active underline color). #} {% macro sub_nav(items, current_path, domain='') %}
{% for it in items %} {% set path = it[0] %} {% set label = it[1] %} {% set count = it[2] if it|length > 2 else None %} {{ label }} {% if count is not none %}{{ count }}{% endif %} {% endfor %}
{% endmacro %} {# Empty state — uniform "nothing yet" panel. #} {% macro empty_state(glyph, message) %}
{{ glyph }} {{ message }}
{% endmacro %} {# Canonical dev sub-nav — SINGLE source of the dev tab list so adding a tab is one edit, not one-per-template. `active_count` shows on the active tab only. Dev-only app: paths carry NO /dev prefix. #} {% macro dev_nav(current_path, active_count=None) %} {% set items = [ ('/', 'Overview'), ('/decisions', 'Decisions'), ('/bugs', 'Bugs'), ('/changelog', 'Changelog'), ('/observations', 'Observations'), ('/targets', 'Targets'), ('/reminders', 'Reminders'), ('/projectrules', 'Rules (PR)'), ('/adjudications', 'Adjudications (Q)'), ('/resolutions', 'Resolutions (RS)'), ('/investigations', 'Investigations (I)'), ('/ideas', 'Ideas'), ('/recognition-keys', 'Recognition Keys'), ('/health', 'Health'), ('/graph', 'Graph'), ] %}
{% for path, label in items %} {{ label }} {% if current_path == path and active_count is not none %}{{ active_count }}{% endif %} {% endfor %}
{% endmacro %} {# Expandable body field — long text (decision/rationale/verdict/evidence/ resolution/findings/description/solution/rule) was unreadable when dumped full into a card. Native
(zero-JS): collapsed shows a one-line preview, click expands the full pre-wrapped text in place. Short text renders plain. #} {% macro field_expand(label, text, preview_len=150, tone='') %} {% if text %}
{{ label }} {% set t = text|string %} {% if t|length > preview_len or '\n' in t %}
{{ t.split('\n')[0][:preview_len] }}{% if t|length > preview_len or '\n' in t %} …{% endif %} {{ t }}
{% else %} {{ t }} {% endif %}
{% endif %} {% endmacro %}
